234 void initIO()
235 {
236  uint8_t Index_len;
237  //all ports are inputs
238  DDRC = 0x00;
239  DDRD = 0x00;
240  DDRB = 0x00;
241  //set port direction to output for ports which are defined as outputs
242  for(Index_len=0; Index_len<EN_NUMBER_OF_ELEMENTS_OUTPUTS; Index_len++)
243  {
244  switch (matchingTableOutputPins_acst[Index_len].portName_en)
245  {
246  case EN_PORT_C:
247  DDRC |= (1 << matchingTableOutputPins_acst[Index_len].portName_en);
248  break;
249  case EN_PORT_B:
250  DDRB |= (1 << matchingTableOutputPins_acst[Index_len].portName_en);
251  break;
252  case EN_PORT_D:
253  DDRD |= (1 << matchingTableOutputPins_acst[Index_len].portName_en);
254  break;
255 
256  default:
257  break;
258 
259  }
260  }
261 
262 
263 
264  /* SETAM TIMERUL 2 PENTRU Modul PHASE CORRECTED PWM*/
265  //setam Timer Counter Control Register A pe None-inverting mode si PWM Phase Corrected Mode
266  TCCR2A |= (1 << COM2A1) | (0 << COM2A0) | (1 << COM2B1)| (0 << COM2B0);
267  // set none-inverting mode on bytes (7,6,5,4) for timer 2
268 
269  TCCR2A |= (0 << WGM21) | (1 << WGM20);
270  // set PWM Phase Corrected Mode using OCR2A TOP value on bytes (1,0)
271 
272  // setam Timer Counter Control Register B pe PWM Phase Corrected Mode si cu un prescaler de 64
273  TCCR2B |= (1 << CS22) | (1 << WGM22);
274  // set prescaler to 64 and starts PWM and sets PWM Phase Corrected Mode
275 }
